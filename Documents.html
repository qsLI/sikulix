import time
def init():
    """ initial state """
    if not exists():
        exit()
    click()
    wait()
    click()

def backHome():
    """ back to city select activity """
    while not exists():
         click()
         wait(, 6)
    statusBar = find().getCenter().offset(0,-64)
    hover(statusBar)


def lottery():
    """ lottery """
    if exists(Pattern().similar(0.91)):
        click(Pattern().similar(0.84))
        try:
            wait(, 6)
        except:
            backHome()
            return
        for x in range(3):
            doubleClick()
            time.sleep(8)
            doubleClick()


def loopCity():
    while True:
        lottery()
        backHome()
        if exists():
            return
        else:
           # slide up
           #wheel(WHEEL_DOWN, 3)
            statusBar = find().getCenter().offset(0,-64)
            wheel(WHEEL_DOWN, 1)
            dragDrop(statusBar, statusBar.offset(Location(10,-100)))

def changeCity(item):
    if exists(Pattern().similar(0.80)):
        click(Pattern().similar(0.80))
        if exists(item):
            click(item)
            wait()
